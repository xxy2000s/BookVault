    01背包问题¶

    f[j]代表当前背包容量为j的时候，可以获取的最大价值。完全背包是从左向右遍历，f[j-V[i]]取到的是拿第i个物品时的值，是新值，可以重复无限的拿，f[j]的值也会随之增加。 V：商品的体积 W：商品的价值

    01背包问题的模版：

    //01背包
for (int i = 0; i < n; i++) {
for (int j = m; j >= V[i]; j--) {
   f[j] = max(f[j], f[j-V[i]] + W[i]);
}
}
//完全背包
for (int i = 0; i < n; i++) {
for (int j = V[i]; j <= m; j++) {
   f[j] = max(f[j], f[j-V[i]] + W[i]);
}
}

//递归法自顶向下
int minDistance(string word1, string word2) {
   int m = word1.size();
   int n = word2.size();
   vector<vector<int> > dp (m+1, vector<int>(n+1));
   for(int i=0;i<=m;++i) dp[i][0] = i;
   for(int j=0;j<=n;++j) dp[0][j] = j;
   for(int i=1;i<=m;++i){
       for(int j=1;j<=n;++j){
           //当前字符串相等 则双指针都前移
           if(word1[i-1]==word2[j-1]) dp[i][j] = dp[i-1][j-1];
           else dp[i][j] = min(min(dp[i-1][j]+1,dp[i][j-1]+1),dp[i-1][j-1]+1);
       }
   }
   return dp[m][n];
}

    class Solution {
    public int rob(int[] nums) {
        if(nums.length == 1) return nums[0];
        if(nums.length == 2) return Math.max(nums[0], nums[1]);
        int[] dp = new int[nums.length];
        dp[0] = nums[0];
        dp[1] = Math.max(nums[0], nums[1]);
        for(int i=2;i<nums.length;i++){
            dp[i] = Math.max(dp[i-2] + nums[i], dp[i-1]);
        }
        return dp[nums.length-1];
    }
}

    403. 青蛙过河¶

    class Solution {
    public boolean canCross(int[] stones) {
        /*
        关注变量：当前stone数组的索引index，跳的步数k
        1.状态定义：dp[idx][k] 代表跳了k步之后来到索引为idx的石头上
        2.状态转移：idx可以从j∈[1, idx-1]位置的石头上转移过来，需要遍历。由于是跳了k步(k = diff = stone[idx]-stone[j])来到idx上，所以跳到前一块石头j的步数必须是 k-1 || k || k+1
                    从而转移方程为 dp[idx][diff] = dp[j][diff-1] || dp[j][diff] || dp[j][diff+1]
        3.初始化：dp[1][1]=true;
        4.遍历顺序：外层遍历i，i∈[2, len-1]，代表stone索引idx。
                    内层遍历j，j∈[1, i-1]，代表从前面的[1, i-1]石头得到的状态转移
        5.返回条件：dp[n-1][k] k∈[1, n-1]  (跳到stone[n-1]的步数可以为[1, n-1])
        */
        int len = stones.length;
        if(stones[1] != 1) return false;
        //dp[n][k]为跳跃了k步来到了stone[n]的位置
        //n对应stone的索引，他的状态应该由i∈[0,n-1]转移而来 由于0、1特殊性可以从[2, n-1]遍历i
        //遍历j，j∈[1, i-1]，diff对应跳跃到当前位置的步数
        //则转移情况等价于dp[i][diff] = dp[j][diff] || dp[j][diff-1] || dp[j][diff+1]
        boolean [][]dp = new boolean[len][len+1];
        dp[1][1] = true;
        //i为stones对应索引，为整体外部遍历
        for(int i=2;i<len;i++){
            //状态转移部分，j∈[1, i-1]为之前的状态遍历
            for(int j=1;j<i;j++){
                //该次所跳步数
                int diff = stones[i] - stones[j];
                if(diff<=j+1){ //j+1为理论最大步数
                    //该次跳diff步，那上一次也只能跳diff-1 || diff || diff+1步，从而得转移方程
                    dp[i][diff] = dp[j][diff-1] || dp[j][diff] || dp[j][diff+1];
                }
            }
        }
        //len是步数k的上限
        for(int k=1;k<len;k++){
            if(dp[len-1][k] == true) return true;
        }
        return false;
    }
}

    887. 鸡蛋掉落¶

    给你 k 枚相同的鸡蛋，并可以使用一栋从第 1 层到第 n 层共有 n 层楼的建筑。

    已知存在楼层 f ，满足 0 <= f <= n ，任何从 高于 f 的楼层落下的鸡蛋都会碎，从 f 楼层或比它低的楼层落下的鸡蛋都不会破。

    每次操作，你可以取一枚没有碎的鸡蛋并把它从任一楼层 x 扔下（满足 1 <= x <= n）。如果鸡蛋碎了，你就不能再次使用它。如果某枚鸡蛋扔下后没有摔碎，则可以在之后的操作中 重复使用 这枚鸡蛋。

    请你计算并返回要确定 f 确切的值 的 最小操作次数 是多少？ 

    class Solution {
    public int superEggDrop(int k, int n) {
        //k 鸡蛋数 n 楼层数 res 需要操作的次数
        //dp[k][n] 为所得结果
        //dp[i-1][j] 为碎了的情况, dp[i][n-j] 为没碎的情况
        //转移方程为 dp[i][j] = max(dp[i-1][j-1], dp[i][n-j])+1
        int dp[][] = new int[k+1][n+1];
        for(int i=1;i<=k;i++){
            for(int j=1;j<=n;j++){
                if(i==1) dp[i][j] = j;
                else{
                    //二分搜索优化得到中间最大值
                    int lo = 1;
                    //顶为j层
                    int hi = j;
                    int res = Integer.MAX_VALUE;
                    while(lo<=hi){
                        //当前搜索的位置为mid层
                        int mid = (lo+hi)/2;
                        int notBroken = dp[i-1][mid-1];
                        int broked = dp[i][j-mid];
                        if(notBroken > broked){
                            hi = mid-1;
                            res = Math.min(res, notBroken+1);
                        }else{
                            lo = mid+1;
                            res = Math.min(res, broked+1);
                        }
                    }
                    dp[i][j] = res;
                }
            }
        }
        return dp[k][n];
    }
}

    时间复杂度O(K * NlogN)

    空间复杂度O(N * K)

    45. 跳跃游戏 II¶

    class Solution {
public:
    int jump(vector<int>& nums) {
        int len = nums.size();
        vector<int> dp(len);
        for(int i=0;i<len;i++) dp[i] = i;
        for(int i=0;i<len;i++){
            for(int j=1;j<=nums[i] && i+j < len;j++){
                dp[i+j] = min(dp[i+j], dp[i]+1);
            }
        }
        return dp[len-1];
    }
};

    class Solution {
public:
    int jump(vector<int>& nums) {
        int start = 0;
        int end = 1;
        int res = 0;
        int maxIdx = 0;
        while(end<nums.size()){
            //从start->end每一个位置都尝试一次来位置最远位置
            for(int i=start;i<end;i++){
                //贪心maxIdx保存当前位置能跳的最远距离
                maxIdx = max(maxIdx, nums[i]+i);
            }
            start = end;
            end = maxIdx+1;
            res++;
        }
        return res;
    }
};

    class Solution {
    public int lengthOfLIS(int[] nums) {
        int res = 1;
        int[] dp = new int[nums.length];
        dp[0] = 1;
        for(int i=1;i<nums.length;i++){
            dp[i] = 1;
            for(int j=0;j<i;j++){
                if(nums[i] > nums[j]){
                    dp[i] = Math.max(dp[i], dp[j]+1);
                }
                res = Math.max(dp[i], res);
            }
        }
        return res;
    }
}

    //贪心+二分
class Solution {
    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        // tail[i]表示所有长度为i+1的最长严格递增子序列的结尾最小值
        int[] tail = new int[n];
        // 初始化长度为1的子序列结尾最小值为nums[0]
        tail[0] = nums[0];
        // 目前的已得到计算得到的子序列长度为1
        int end = 0;
        for (int i = 1; i < n; i++) {
            // 如果nums[i]比tail[end]还大，那么将其放到tail[end]后面，可以变为end+2的递增子序列
            if (tail[end] < nums[i]) {
                tail[++end] = nums[i];
            // 否则，找到大于等于nums[i]的tail[j]，缩小它即可缩小长度为j+1的最长严格递增子序列的结尾最小值，便于找到更长的序列
            } else {
                int left = 0;
                int right = end;
                while (left < right) {
                    int mid = left + (right - left) / 2;
                    if (tail[mid] >= nums[i]) {
                        right = mid;
                    } else {
                        left = mid + 1;
                    }
                }
                // 更新tail[j]
                tail[left] = nums[i];
            }
        }
        // 最长的递增子序列长度
        return end + 1;
    }
}

    class Solution {
    public int uniquePaths(int m, int n) {
        if(m<1 || n<1) return 0;
        if(m == 1 || n == 1) return 1;
        int[][] dp = new int[m][n];
        for(int i=1;i<m;i++){
            for(int j=1;j<n;j++){
                dp[i][j] = dp[i-1][j] + dp[i][j-1];
            }
        }
        return dp[m-1][n-1];
    }
}

    63. 不同路径 II¶

    class Solution {
    public int uniquePathsWithObstacles(int[][] obstacleGrid) {
        int m = obstacleGrid.length;
        int n = obstacleGrid[0].length;
        if(obstacleGrid[0][0] == 1) return 0;
        int[][] dp = new int[m][n];
        dp[0][0] = 1;
        //第一行和第一列一旦碰到障碍则后面的dp[i][0] dp[0][j] 一定要置0，要跳出for循环
        for(int i=0;i<m && obstacleGrid[i][0] == 0;i++){
            dp[i][0] = 1;
        }
        for(int i=0;i<n && obstacleGrid[0][i] == 0;i++){
            dp[0][i] = 1;
        }
        for(int i=1;i<m;i++){
            for(int j=1;j<n;j++){
                if(obstacleGrid[i][j] == 0) dp[i][j] = dp[i-1][j]+dp[i][j-1];
                else dp[i][j] = 0;
            }
        }
        return dp[m-1][n-1];
    }
}

    64. 最小路径和¶

    输出路径的话，使用dpmn往前反推，dp(i)(j) - grid(i)(j)再和左边和上边比较

    //O(mn) O(mn)
class Solution {
    public int minPathSum(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        if(m == 0 || n == 0) return 0;
        int[][] dp = new int[m][n];
        dp[0][0] = grid[0][0];
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                if(i == 0 && j == 0) continue;
                else if(i==0) dp[i][j] = dp[i][j-1] + grid[i][j];
                else if(j==0) dp[i][j] = dp[i-1][j] + grid[i][j];
                else dp[i][j] = Math.min(dp[i-1][j], dp[i][j-1]) + grid[i][j];
            }
        }
        return dp[m-1][n-1];
    }
}

class Solution {
    public int minDistance(String word1, String word2) {
        //二维DP Table 两个字符串都补上空
        //m为行数
        int m = word1.length();
        int n = word2.length();
        if(m*n == 0) return m+n;
        int[][] dp = new int[m+1][n+1];
        dp[0][0] = 0;
        for(int i=0;i<=m;i++){
            for(int j=0;j<=n;j++){
                //对第一行第一列进行初始化
                if(i==0){
                    dp[i][j] = j;
                }else if(j==0){
                    dp[i][j] = i;
                }else if(word1.charAt(i-1)==word2.charAt(j-1)){
                    dp[i][j] = dp[i-1][j-1];
                }else{
                    //dp[i-1][j-1]替换
                    //dp[i-1][j]上边删除
                    //dp[i][j-1]左边插入
                    dp[i][j] = Math.min(dp[i-1][j-1], Math.min(dp[i-1][j], dp[i][j-1]))+1;
                }
            }
        }
        return dp[m][n];
    }

    //复杂度优化
public int minDistance(String word1, String word2) {
    int m = word1.length();
    int n = word2.length();
    int[] dp = new int[n+1];
    for (int i = 1; i <= n; i++) {
        dp[i] = i;
    }
    for (int i = 1; i <= m; i++) {
        int prev = dp[0];
        dp[0] = i;
        for (int j = 1; j <= n; j++) {
            int temp = dp[j];
            if (word1.charAt(i-1) == word2.charAt(j-1)) {
                dp[j] = prev;
            } else {
                int insert = prev + 1;
                int delete = dp[j] + 1;
                int replace = dp[j-1] + 1;
                dp[j] = Math.min(insert, Math.min(delete, replace));
            }
            prev = temp;
        }
    }
    return dp[n];
}

    //复杂度优化
class Solution {
    public int longestCommonSubsequence(String text1, String text2) {
        int l1 = text1.length();
        int l2 = text2.length();
        int dp[] = new int[l2+1];
        for(int i=1;i<=l1;i++){
            //pre就相当左上角的dp[i-1][j-1]
            int pre = 0;
            for(int j=1;j<=l2;j++){
                //dp[j]要被改变,所以暂存作为左上角的dp[i-1][j-1]并之后得赋予给pre。
                int tmp = dp[j];
                //dp[j]这时候被更新计算了，所以不一定还是之前的dp[i-1][j-1]
                if(text1.charAt(i-1) == text2.charAt(j-1)) dp[j] = pre+1;
                else dp[j] = Math.max(dp[j], dp[j-1]);
                //暂存左上角dp[i-1][j-1]供下一轮循环使用
                pre = tmp;
            }
        }
        return dp[l2];
    }
}

    class Solution {

        int l2 = nums2.length;
        int dp[] = new int[l2+1];

                if(nums1[i-1] == nums2[j-1]) dp[j] = dp[j-1]+1;
                else dp[j] = 0;
                res = Math.max(res, dp[j]);
            }
        }
        return res;
    }

    class Solution {
public:
    int longestValidParentheses(string s) {
        stack<char> st;
        int res = 0;
        //dp[i]表示以i结尾匹配到的最长结果
        vector<int> dp(s.size(), 0);
        for(int i=0;i<s.size();i++){
            //见左括号一律入栈
            if(s[i]=='(') st.push(s[i]);
            else{
                if(st.empty()) continue;
                //弹出右括号
                else{
                    st.pop();
                    //如果是类似(())这种嵌套，则要考虑s[i-1]情况
                    if(s[i-1]==')') dp[i] = dp[i-1]+2;
                    else dp[i] = 2;
                    //i-dp[i]表示该匹配范围前面的情况,如果是()(())则能匹配到前面的类型
                    if(i-dp[i]>=0 && s[i-dp[i]]==')') dp[i]+=dp[i-dp[i]];
                    res = max(res, dp[i]);
                }
            }
        }
        return res;
    }
};

    背包问题¶

    注意背包问题和凑零钱问题的区别；具体体现在初始化以及条件判断情形

    **背包问题：**能装下就行，不要求正好凑满，所以不可达的数可以初始化为初始value或者0

    **凑零钱1：**正好凑到amout数，求最少硬币，过程中取min，所以不可达的数要初始化为INT_MAX

    凑零钱2：正好到amout数，求方法数，所以是+=，而不是取max或者min

    01背包中二维dp数组的两个for遍历的先后循序是可以颠倒了，一维dp数组的两个for循环先后循序一定是先遍历物品，再遍历背包容量。

    在完全背包中，对于一维dp数组来说，其实两个for循环嵌套顺序是无所谓的

    0-1背包¶

    //二维dp
public class Solution{
    public int solveBag(int[] weight, int[] value, int bagweight){
        if(weight.length < 1) return 0;
        int[][] dp = new int[weight.length][bagweight+1];
        //注意初始化
        for(int i=bagweight;i>=weight[0];i--){
            dp[0][i] = dp[0][i-weight[0]]+value[0];
        }
        for(int i=0;i<weight.length;i++){
            for(int j=bagweight;j>=0;j--){
                //看能否加入
                if(j<weight[i]) dp[i][j] = dp[i-1][j];
                else dp[i][j] = Math.max(dp[i-1][j], dp[i-1][j-weight[i]]+value[i]);
            }
        }
        return dp[weight.length-1][bagweight];
    }
}

        }
        return dp[bagweight];
    }
}

    322. 零钱兑换¶

    本题求钱币最小个数，那么钱币有顺序和没有顺序都可以，都不影响钱币的最小个数

    所以本题并不强调集合是组合还是排列。

    如果求组合数就是外层for循环遍历物品，内层for遍历背包。

    如果求排列数就是外层for遍历背包，内层for循环遍历物品

    class Solution {
    public int coinChange(int[] coins, int amount) {
        int[] dp = new int[amount+1];
        Arrays.fill(dp, Integer.MAX_VALUE);
        //需要对dp[0]初始化，否则后面无法正确递推
        dp[0] = 0;
        for(int i=0;i<coins.length;i++){
            for(int j=coins[i];j<=amount;j++){
                if(dp[j-coins[i]] != Integer.MAX_VALUE){
                    //这里是求最少的钱币个数，所以是初始化为max，然后取min
                    dp[j] = Math.min(dp[j-coins[i]]+1, dp[j]);
                }
            }
        }
        if(dp[amount] == Integer.MAX_VALUE) return -1;
        else return dp[amount];
    }
}

    class Solution {
public:
    int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {
        int start = 0;
        //用于局部判断油量是否够,判断能否从i->i+1,不能则重置起点
        int run = 0;
        //用于全局判断油量是否够,因为是绕圈,所以全局都要计算不影响
        int rest = 0;
        for(int i=0;i<gas.size();i++){
            run += gas[i]-cost[i];
            rest += gas[i]-cost[i];
            if(run<0){
                start = i+1;
                run = 0;
            }
        }
        return rest>=0?start:-1;
    }
};
