public TreeNode buildTree(Object[] nums){
    List<TreeNode> list = new ArrayList<>();
    int len = nums.length;
    if(len==0) return null;
    TreeNode root = null;
    //将所有树节点存入List中
    for(int i=0;i<len;i++){
        TreeNode cur = null;
        if(nums[i]!=null){
            cur = new TreeNode(nums[i]);
        }
        list.add(cur);
        if(i==0) root = cur;
    }
    //依据List构建出树
    for(int i=0;i*2+1<n;i++){
        TreeNode cur = list.get(i);
        //不为空的话加上左右节点
        if(cur!=null){
            cur.left = list.get(2*i+1);
            if(i*2+2<n){
                cur.right = list.get(2*i+2);
            }
        }
    }
    return root;
}

    struct TreeNode{
    int val;
    TreeNode* left;
    TreeNode* right;
    TreeNode():val(0),left(nullptr),right(nullptr){}
    TreeNode(int _val):val(_val),left(nullptr),right(nullptr){}
}

TreeNode* buildTree(vector<int>& nums){
    if(nums.size()==0) return nullptr;
    vector<TreeNode*> tree;
    for(int i=0;i<nums.size();i++){
        TreeNode* tmp = nullptr;
        if(nums[i]!=-1){
            tmp = new TreeNode(nums[i]);
        }
        tree.push(tmp);
    }
    TreeNode* root = tree[0];
    for(int i=0;i*2+2<tree.size();i++){
        TreeNode* cur = tree[i];
        if(cur!=nullptr){
            cur->left = tree[i*2+1];
            cur->right = tree[i*2+2];
        }
    }
    return root;
}

            int mid = left + ((right-left)>>1);
            //先判中间节点

                }else{
                    left = mid+1;
                }
            }
        }

    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        // tail[i]表示所有长度为i+1的最长严格递增子序列的结尾最小值
        int[] tail = new int[n];
        // 初始化长度为1的子序列结尾最小值为nums[0]
        tail[0] = nums[0];
        // 目前的已得到计算得到的子序列长度为1
        int end = 0;
        for (int i = 1; i < n; i++) {
            // 如果nums[i]比tail[end]还大，那么将其放到tail[end]后面，可以变为end+2的递增子序列
            if (tail[end] < nums[i]) {
                tail[++end] = nums[i];
            // 否则，找到大于等于nums[i]的tail[j]，缩小它即可缩小长度为j+1的最长严格递增子序列的结尾最小值，便于找到更长的序列
            } else {
                int left = 0;
                int right = end;
                while (left < right) {
                    int mid = left + (right - left) / 2;
                    if (tail[mid] >= nums[i]) {
                        right = mid;
                    } else {
                        left = mid + 1;
                    }
                }
                // 更新tail[j]
                tail[left] = nums[i];
            }
        }
        // 最长的递增子序列长度
        return end + 1;
    }
}

            run += gas[i]-cost[i];
            rest += gas[i]-cost[i];
            if(run<0){
                start = i+1;
                run = 0;
            }
        }

    百度面到问有一个randP(),有P的概率生成0，1-P的概率生成1，求问怎么生成randN（）

    连续randP两次，如果是(0,0)或者(1,1)则抛弃，然后规定(0,1)的结果为0,(1,0)的结果为1，这样你就可以等概率得到0,1了，那么相当于你得到了一个等概率rand2的方法，这个时候N就可以用rand2去拼了

    class Solution extends SolBase {
    public int rand10() {
        int res = 49;
        while(res>40){
            res = (rand7()-1)*7+rand7();
        }
        return res%10+1;
    }
}

    string multiply(string s1, string s2){
    int len1 = s1.size();
    int len2 = s2.size();
    if(len1 ==0 || len2 == 0 || s1=="0" || s2=="0") return "0";
    string res(len1+len2, '0');
    for(int i=len2-1;i>=0;i--){
      //进位
      int step = 0;
      for(int j=len1-1;j>=0;j--){
          //位乘积结果
          int mul = (s2[i]-'0')*(s1[j]-'0');
          //当前位总和，包含乘积结果和之前的进位
          int sum = res[i+j+1]-'0' + step + mul%10;
          //计算当前位结果
          res[i+j+1] = sum%10 + '0';
          //得出进位
          step = sum/10 + mul/10;
      }
      res[i] += step;
    }
    for(int i=0;i<res.size();i++){
        if(res[i] != '0') return res.substr(i);
    }
    return "0";
}
